70void Widget::open_table(){
71 QFile file(path_table);
72 if (file.open(QIODevice::ReadOnly | QIODevice::Text)){
73 model->clear();
74
75 QTextStream in(&file);
76
77 QString headerLine = in.readLine();
78 QStringList headerFields = headerLine.split(",");
79 model->setHorizontalHeaderLabels(headerFields);
80 for (int64_t i = 0; i < headerFields.size(); i ++) {
81 if (headerFields[i] == "diagnosis") {
82 col_diagnosis_at = i;
83 }
84 }
85
86 int row = 0;
87 while (!in.atEnd()) {
88 QString line = in.readLine();
89 QStringList fields = line.split(",");
90
91 for (int col = 0; col < fields.size(); col++){
92 QString& val = fields[col];
93// benign 良性 0
94// malignant 恶性 1
95 if (val == "B"){
96 val = "0";
97 }
98 else if (val == "M"){
99 val = "1";
100 }
101 QStandardItem *item = new QStandardItem(fields[col]);
102 model->setItem(row, col, item);
103 }
104
105 row++;
106 }
107
108 file.close();
109 }
110 else{
111 qDebug() << "fail to open";
112 }

243std::vector<int> Widget::get_labels_of(Cluster_method method){
244 const size_t cnt_total_rows = model->rowCount();
245 size_t cnt_total_cols = model->columnCount();
246 size_t col_cluster = cnt_total_cols;
247 bool exist_cluster = false;
248 for (int col = cnt_total_cols - 1; col >= 0; col --){
249 if (model->horizontalHeaderItem(col)->text() == map_method_string[method]){
250 col_cluster = col;
251 exist_cluster = true;

257 map_cluster_groups[method] = 0;

270 max_label = labels.back();

275// QMessageBox::critical(this, "Error", "没有找到该聚类分组数据。");

372void Widget::on_button_scatter_clicked()
373{
374 // 获取选中的单元格模型
375 QItemSelectionModel *selectionModel = ui->tableView->selectionModel();
376
377 // 获取选中的列
378 QModelIndexList selectedColumns = selectionModel->selectedColumns();

382 QMessageBox::critical(this, "Error", "Please select only 2 columns.");
383 return;
384 }
385
386 // 创建两个vector<float>来存储选中列的数据
387 std::vector<float> dataX;
388 std::vector<float> dataY;
389
390 const int colX = selectedColumns[0].column();
391 const int colY = selectedColumns[1].column();
392
393 if (colX == colY) {
394 QMessageBox::critical(this, "Error", "Please select 2 different columns.");
395 return;
396 }
397
398 if (model->horizontalHeaderItem(colX)->text() == "id" ||
399 model->horizontalHeaderItem(colY)->text() == "id") {
400 QMessageBox::critical(this, "Error", "Please do not select id column.");
401 return;
402 }
403
404 if (model->horizontalHeaderItem(colX)->text() == "diagnosis" ||
405 model->horizontalHeaderItem(colY)->text() == "diagnosis") {
406 QMessageBox::critical(this, "Error", "Please do not select diagnosis column.");
407 return;
408 }
409
410 const auto header = ui->tableView->horizontalHeader();
411 const QString headerX = header->model()->headerData(colX, Qt::Horizontal).toString();
412 const QString headerY = header->model()->headerData(colY, Qt::Horizontal).toString();
413
414 auto get_data = [&](const int col){
415 for (int row = 0; row < model->rowCount(); row++){
416 QModelIndex index = model->index(row, col);
417 QVariant data = model->data(index);
418
419 bool conversionOk;
420 float value = data.toFloat(&conversionOk);
421 if (conversionOk) {
422 if (index.column() == colX) {
423 dataX.push_back(value);
424 } else if (index.column() == colY) {
425 dataY.push_back(value);
426 }
427 }
428 }
429 };
430
431 get_data(colX);
432 get_data(colY);
433
434// 打开新窗口
435 auto window_scatter = new Window_Scatter(dataX, dataY, headerX, headerY, this);
436 window_scatter->show();
