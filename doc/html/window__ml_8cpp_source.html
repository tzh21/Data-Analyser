148void Window_ML::on_button_ratio_clicked()
149{
150 // 获取数据集和训练集的比例
151 float ratio = comb_ratio->currentText().toFloat();
152 int cnt_test = static_cast<int>(diagnosis.size() * ratio);
153 // 检查测试集数量是否为0
154 if (cnt_test <= 0){
155 QMessageBox::critical(this, "错误", "测试集数量为0");
156 return;
157 }
158
159 // 划分训练集和测试集
160 std::vector<std::pair<int, int>> col_diagnosis;
161 for (size_t i = 0; i < diagnosis.size(); ++i)
162 {
163 col_diagnosis.push_back(std::make_pair(i, diagnosis[i]));
164 }
165 std::srand(static_cast<unsigned int>(std::time(nullptr)));
166 std::random_shuffle(col_diagnosis.begin(), col_diagnosis.end());
167 std::vector<std::pair<int, int>> train;
168 std::vector<std::pair<int, int>> test;
169 idx_test.clear();
170 for (size_t i = 0; i < cnt_test; i ++){
171 test.push_back(col_diagnosis[i]);
172 idx_test.push_back(col_diagnosis[i].first);
173 }
174 idx_train.clear();
175 for (size_t i = cnt_test; i < col_diagnosis.size(); i ++){
176 train.push_back(col_diagnosis[i]);
177 idx_train.push_back(col_diagnosis[i].first);
178 }
179
180 // 绘制训练集表格
181 table_train->clear();
182 table_train->setRowCount(train.size());
183 table_train->setColumnCount(2);
184 table_train->setHorizontalHeaderLabels(QStringList() << "index" << "diagnosis");
185 for (size_t row = 0; row < train.size(); ++row)
186 {
187 table_train->setItem(row, 0, new QTableWidgetItem(QString::number(train[row].first)));
188 table_train->setItem(row, 1, new QTableWidgetItem(QString::number(train[row].second)));
189 }
190 // 绘制测试集表格
191 table_test->clear();
192 table_test->setRowCount(test.size());
193 table_test->setColumnCount(3);
194 table_test->setHorizontalHeaderLabels(QStringList() << "index" << "diagnosis" << "predict");
195 for (size_t row = 0; row < test.size(); ++row)
196 {
197 table_test->setItem(row, 0, new QTableWidgetItem(QString::number(test[row].first)));
198 table_test->setItem(row, 1, new QTableWidgetItem(QString::number(test[row].second)));
199 table_test->setItem(row, 2, new QTableWidgetItem(QString::number(-1)));
200 }

206void Window_ML::on_button_train_clicked(){
207 if (samples.empty() || samples[0].empty()){
208 QMessageBox::critical(this, "错误", "样本为空");
209 return;
210 }
211 const int size_features = samples[0].size();
212 const int size_samples = samples.size();
213 const int size_train_samples = idx_train.size();
214 const int size_test_samples = idx_test.size();
215 const int size_diagnosis = diagnosis.size();
216 if (size_diagnosis != size_samples || idx_test.size() == 0 || idx_train.size() == 0){
217 QMessageBox::critical(this, "错误", "数据集和标签数量不一致");
218 return;
219 }
220
221 // 将数据集和标签转换为xgboost的数据结构
222 unsigned int label_train[size_train_samples];
223 for (int i = 0; i < size_train_samples; i ++){
224 label_train[i] = diagnosis[idx_train[i]];
225 }
226 float train[size_train_samples][size_features];
227 for (int i = 0; i < size_train_samples; i ++){
228 for (int j = 0; j < size_features; j ++){
229 train[i][j] = samples[idx_train[i]][j];
230 }
231 }
232
233 DMatrixHandle dtrain;
234
235 XGDMatrixCreateFromMat((float *) train, size_train_samples, size_features, -1, &dtrain);
236
237 XGDMatrixSetUIntInfo(dtrain, "label", label_train, size_train_samples); // set label parameter
238
239 // Create booster
240 BoosterHandle booster;
241 XGBoosterCreate(&dtrain, 1, &booster);
242
243 // 设置booster特征名称
244 const char *fnames[size_features];
245 for (int i = 0; i < size_features; i ++){
246 fnames[i] = feature_names[i].c_str();
247 }
248 XGBoosterSetStrFeatureInfo(booster, "feature_name", fnames, size_features);
249
250 // 检查参数
251 bool is_valid = true;
252 int max_depth = edit_max_depth->text().toInt(&is_valid);
253 if (!is_valid){
254 QMessageBox::critical(this, "错误", "最大深度不是整数");
255 return;
256 }
257 std::string str_max_depth = std::to_string(max_depth);
258 const char *c_str_max_depth = str_max_depth.c_str();
259 int iter = edit_iter->text().toInt(&is_valid);
260 if (!is_valid){
261 QMessageBox::critical(this, "错误", "迭代次数不是整数");
262 return;
263 }
264
265 // Set parameters
266 XGBoosterSetParam(booster, "objective", "reg:squarederror");
267 XGBoosterSetParam(booster, "max_depth", c_str_max_depth);
268 XGBoosterSetParam(booster, "eta", "0.1");
269
270 // Train for 10 iterations
271 for (int round = 0; round < iter; round++) {
272 XGBoosterUpdateOneIter(booster, round, dtrain);
273 }
274
275 // 获取特征贡献度
276 char const config[] =
277 "{\"importance_type\": \"weight\"}";
278 bst_ulong out_n_features;
279 const char** out_features = new const char*[size_features];
280 bst_ulong out_dim;
281 const int max_n_features = 100;
282 bst_ulong const* out_shape = new bst_ulong[max_n_features];
283 float const* out_scores = new float[size_features];
284 XGBoosterFeatureScore(
285 booster, config,
286 &out_n_features, &out_features,
287 &out_dim, &out_shape,
288 &out_scores);
289
290 // 排序后存储特征贡献度
291 feature_importance.clear();
292 qDebug() << size_features;
293 for (int i = 0; i < out_n_features; i ++){
294 feature_importance.push_back(std::make_pair(std::string(out_features[i]), out_scores[i]));
295 }
296 std::sort(feature_importance.begin(), feature_importance.end(), [](const std::pair<std::string, float> &a, const std::pair<std::string, float> &b){
297 return a.second < b.second;
298 });
299
300 // 预测测试集
301
302 // 转换C数据结构
303 unsigned int label_test[size_test_samples];
304 for (int i = 0; i < size_test_samples; i ++){
305 label_test[i] = diagnosis[idx_test[i]];
306 }
307 float test[size_test_samples][size_features];
308 for (int i = 0; i < size_test_samples; i ++){
309 for (int j = 0; j < size_features; j ++){
310 test[i][j] = samples[idx_test[i]][j];
311 }
312 }
313 DMatrixHandle dtest;
314 XGDMatrixCreateFromMat((float *) test, size_test_samples, size_features, -1, &dtest);
315 XGDMatrixSetUIntInfo(dtest, "label", label_test, size_test_samples);
316
317 bst_ulong out_len;
318 const float* out_result;
319 XGBoosterPredict(booster, dtest, 0, 2, 0, &out_len, &out_result);
320
321 // 将预测结果写入表格
322 for (size_t row = 0; row < out_len; ++row)
323 {
324 if (!table_test->item(row, 2)){
325 QMessageBox::critical(this, "错误", "测试集表格中有空值");
326 return;
327 }
328 if (out_result[row] < 0.5){
329 table_test->item(row, 2)->setText("0");
330 }
331 else {
332 table_test->item(row, 2)->setText("1");
333 }
334 }
335
336 // 统计混淆矩阵
337 for (int row = 0; row < table_test->rowCount(); row ++){
338 if (!table_test->item(row, 1) || !table_test->item(row, 2)){
339 QMessageBox::critical(this, "错误", "测试集表格中有空值");
340 return;
341 }
342 if (table_test->item(row, 1)->text() == "0"){
343 if (table_test->item(row, 2)->text() == "0"){
344 cnt_true_negative ++;
345 }
346 else {
347 cnt_false_positive ++;
348 }
349 }
350 else {
351 if (table_test->item(row, 2)->text() == "0"){
352 cnt_false_negative ++;
353 }
354 else {
355 cnt_true_positive ++;
356 }
357 }
358 }
359
360 // 计算F1-score
361 float precision = cnt_true_positive / (cnt_true_positive + cnt_false_positive);
362 float recall = cnt_true_positive / (cnt_true_positive + cnt_false_negative);
363 float f1_score = 2 * (precision * recall) / (precision + recall);
364 edit_f1score->setText(QString::number(f1_score));
365
366 // 计算AUC
367 // 计算ROC曲线的数据点
368 std::vector<float> true_positive_rates;
369 std::vector<float> false_positive_rates;
370
371 for (float threshold = 0.0; threshold <= 1.0; threshold += 0.01) {
372 float true_positives = 0;
373 float false_positives = 0;
374 float false_negatives = 0;
375 float true_negatives = 0;
376
377 for (bst_ulong i = 0; i < out_len; i++) {
378 float predicted_value = out_result[i];
379 int true_label = label_test[i];
380
381 if (predicted_value >= threshold && true_label == 1) {
382 true_positives++;
383 } else if (predicted_value >= threshold && true_label == 0) {
384 false_positives++;
385 } else if (predicted_value < threshold && true_label == 1) {
386 false_negatives++;
387 } else if (predicted_value < threshold && true_label == 0) {
388 true_negatives++;
389 }
390 }
391
392 // 预测阳性在所有阳性中所占的比例
393 float true_positive_rate = true_positives / (true_positives + false_negatives);
394 // 预测阳性在所有阴性中所占的比例
395 float false_positive_rate = false_positives / (false_positives + true_negatives);
396
397 true_positive_rates.push_back(true_positive_rate);
398 false_positive_rates.push_back(false_positive_rate);
399 }
400
401 // 计算AUC，即ROC曲线下的面积
402 float auc = 0.0;
403 for (size_t i = 1; i < true_positive_rates.size(); i++) {
404 float delta_fpr = false_positive_rates[i - 1] - false_positive_rates[i];
405 auc += true_positive_rates[i - 1] * delta_fpr;
406 }
407
408 edit_auc->setText(QString::number(auc));
409
410 // 释放内存
411 XGBoosterFree(booster);
412 XGDMatrixFree(dtrain);
413 XGDMatrixFree(dtest);
414 // delete[] out_features;
415 // delete[] out_shape;
416 // delete[] out_scores;
417}
